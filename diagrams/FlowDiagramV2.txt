    A((Start))
    B{User enters requirement}
    C[API Gateway receives request]
    D[AI Orchestrator builds MCP]
    E{Call LLM Provider}
    F[Receive structured JSON]
    G[Publish Job to Queue]

    M[Record Artifacts to Storage]
    N[Report Final Result]
    O((End))

    G --> H
    H --> I
    I --> J
    J -- No --> K
    J -- Yes --> L
    L --> K
    K --> M
    M --> N
    N --> O
    P --> O
